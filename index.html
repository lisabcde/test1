async function api(action, payload) { saveConfig(); const apiUrl = $('apiUrl').value.trim(); const apiKey = $('apiKey').value.trim(); if (!apiUrl) throw new Error('請先填 API URL'); if (!apiKey) throw new Error('請先填 API Key'); if (action === 'lookup') { const url = new URL(apiUrl); url.searchParams.set('action', 'lookup'); url.searchParams.set('key', apiKey); url.searchParams.set('barcode', payload.barcode || ''); const res = await fetch(url.toString(), { method: 'GET' }); const text = await res.text(); log('API raw response: ' + text); return JSON.parse(text); } if (action === 'move') { const form = new URLSearchParams(); form.append('action', 'move'); form.append('key', apiKey); form.append('barcode', payload.barcode || ''); form.append('delta', String(payload.delta || 0)); form.append('note', payload.note || ''); const res = await fetch(apiUrl, { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded;charset=UTF-8' }, body: form.toString() }); const text = await res.text(); log('API raw response: ' + text); return JSON.parse(text); } throw new Error('unknown action'); }
